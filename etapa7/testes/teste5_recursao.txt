// Funcao recursiva simples - countdown
int countdown(int c1)
{
  if (c1 > 0)
  {
    print c1;
    param = c1 - 1;
    resultado = countdown(param);
  }
  return 0;
}

// Fatorial iterativo
int fatorialIter(int f1)
{
  resultado = 1;
  i = 1;
  while (i <= f1)
  {
    resultado = resultado * i;
    i = i + 1;
  }
  return resultado;
}

// Fibonacci iterativo
int fibonacciIter(int fb1)
{
  a = 0;
  b = 1;
  i = 0;
  while (i < fb1)
  {
    temp = a + b;
    a = b;
    b = temp;
    i = i + 1;
  }
  return a;
}

// Potencia: calcula base^8
int potencia(int p1)
{
  resultado = p1 * p1;
  resultado = resultado * resultado;
  resultado = resultado * resultado;
  return resultado;
}

// Soma iterativa de 1 a n
int somaIter(int s1)
{
  resultado = 0;
  i = 1;
  while (i <= s1)
  {
    resultado = resultado + i;
    i = i + 1;
  }
  return resultado;
}

int dobro(int d1)
{
  return d1 + d1;
}

int triplo(int t1)
{
  return t1 + t1 + t1;
}
