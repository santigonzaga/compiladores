// TESTE 1: Operacoes Basicas
// Testa: variaveis globais, atribuicao, aritmetica, print

int a = 10;
int b = 5;
int c = 0;

int main()
{
  // Soma
  c = a + b;
  print "a + b = ";
  print c;

  // Subtracao
  c = a - b;
  print "a - b = ";
  print c;

  // Multiplicacao
  c = a * b;
  print "a * b = ";
  print c;

  // Divisao
  c = a / b;
  print "a / b = ";
  print c;
}
